% define kanji command ---------------------------------------------------------
%\kanji{KanJi}{on\\kun}{stroke}{radical}{radical-name}{KanJi-meaning}
\newcommand{\kanjii}[6]{
    \begin{tcbraster}[
        raster column skip=1pt,
        raster row skip  =1pt, 
        raster force size=false, 
        raster columns=4, raster rows=1]
        \kbox{#1}
        \begin{onkun}
            \par \vspace{6\zw} \hspace{1\zw} 
            \fcbox{\rmfamily\tatechuyoko{#3}画}
            \par
            \vspace{-7.5\zw}#2
        \end{onkun} 
        \begin{bushu}
            #4 \tcblower \center #5
        \end{bushu}
        \fcbox{\rmfamily\tatechuyoko{#6}}
    \end{tcbraster}
}
